### StatusBadge Component

```typescript
// frontend/src/components/Dashboard/StatusBadge.tsx

import React from 'react';
import { CheckCircle, AlertTriangle, RefreshCw, Wrench, Star, XCircle } from 'lucide-react';
import styled from 'styled-components';

interface StatusBadgeProps {
  status: 'real' | 'mock' | 'partial' | 'fix' | 'progress' | 'new' | 'error';
  compact?: boolean;
}

const BadgeContainer = styled.span<{ status: string; compact: boolean }>`
  display: inline-flex;
  align-items: center;
  gap: 0.25rem;
  padding: ${props => props.compact ? '0.125rem 0.375rem' : '0.25rem 0.5rem'};
  border-radius: 4px;
  font-size: ${props => props.compact ? '0.625rem' : '0.75rem'};
  font-weight: 600;

  ${props => {
    switch (props.status) {
      case 'real':
        return `
          background: rgba(16, 185, 129, 0.2);
          color: #10b981;
          border: 1px solid #10b981;
        `;
      case 'mock':
        return `
          background: rgba(245, 158, 11, 0.2);
          color: #f59e0b;
          border: 1px solid #f59e0b;
        `;
      case 'partial':
      case 'progress':
        return `
          background: rgba(59, 130, 246, 0.2);
          color: #3b82f6;
          border: 1px solid #3b82f6;
        `;
      case 'fix':
      case 'error':
        return `
          background: rgba(239, 68, 68, 0.2);
          color: #ef4444;
          border: 1px solid #ef4444;
        `;
      case 'new':
        return `
          background: rgba(0, 255, 255, 0.2);
          color: #00ffff;
          border: 1px solid #00ffff;
        `;
      default:
        return '';
    }
  }}
`;

const statusConfig = {
  real: { icon: CheckCircle, label: 'Real' },
  mock: { icon: AlertTriangle, label: 'Mock' },
  partial: { icon: RefreshCw, label: 'Partial' },
  progress: { icon: RefreshCw, label: 'WIP' },
  fix: { icon: Wrench, label: 'Fix' },
  new: { icon: Star, label: 'New' },
  error: { icon: XCircle, label: 'Error' },
};

export const StatusBadge: React.FC<StatusBadgeProps> = ({ status, compact = false }) => {
  const config = statusConfig[status];
  const Icon = config.icon;

  return (
    <BadgeContainer status={status} compact={compact}>
      <Icon size={compact ? 10 : 12} />
      {!compact && <span>{config.label}</span>}
    </BadgeContainer>
  );
};
```
